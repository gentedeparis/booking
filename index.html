function renderDossierContent(guia){ const entries = dossierEntries[guia] || []; const el = document.getElementById('dossierContent'); // Calcular totales let totalCreditos = 0; let pendienteCreditos = 0; let cobradoCreditos = 0; entries.forEach(e => { const cred = parseFloat(e.creditos) || 0; totalCreditos += cred; if(e.estadoPago === 'pagado') { cobradoCreditos += cred; } else { pendienteCreditos += cred; } }); // Generar HTML del score mensual let scoreHtml = `
Score mensual
${totalCreditos} cr.
TOTAL
${totalCreditos}
PENDIENTE
${pendienteCreditos}
COBRADO
${cobradoCreditos}
`; const byMonth = {}, mesMap = {}; entries.filter(e=>e.fecha).forEach(e=>{ e.fecha = normalizarFecha(e.fecha); const p = e.fecha.split('-'), mk = p[0]+'-'+p[1]; if(!byMonth[mk]) byMonth[mk] = []; byMonth[mk].push(e); if(!mesMap[mk]){ const last = new Date(parseInt(p[0]),parseInt(p[1]),0).getDate(); mesMap[mk] = { inicio: p[0]+'-'+p[1]+'-01', fin: p[0]+'-'+p[1]+'-'+String(last).padStart(2,'0'), total:0, pagado:0 }; } mesMap[mk].total += parseFloat(e.creditos)||0; if(e.estadoPago === 'pagado') mesMap[mk].pagado += parseFloat(e.creditos)||0; }); const monthNames = ['Enero','Febrero','Marzo','Abril','Mayo','Junio','Julio','Agosto','Septiembre','Octubre','Noviembre','Diciembre']; let html = `
📋 Planing & Créditos — ${escH(guia)}
`; html += scoreHtml; html += `
TourFecha/HoraIdiomaPAX/Asist.Créd.Estado
`; Object.keys(byMonth).sort().reverse().forEach(mk=>{ const isAct = mk === new Date().getFullYear()+'-'+String(new Date().getMonth()+1).padStart(2,'0'); const acId = 'da-'+mk.replace('-','_'); const mesTotal = mesMap[mk]?.total || 0; const mesPagado = mesMap[mk]?.pagado || 0; const mesPendiente = mesTotal - mesPagado; html += `
${monthNames[parseInt(mk.split('-')[1])-1]} ${mk.split('-')[0]}
${mesPendiente} pend. · ${mesPagado} cob.
${mesTotal} cr.${isAct?'▲':'▼'}
`; byMonth[mk].sort((a,b)=>a.fecha>b.fecha?-1:1).forEach(e=>{ const ep = e.estadoPago||'pendiente', ip = ep==='pagado', esInforme = ep==='informe_pendiente'; html += `
${escH(e.tourNombre||'')}${esInforme?'📥':''}
${formatDDMMYYYY(e.fecha)}${e.horario?' '+e.horario:''}
${e.horario||'--'}
`; }); html += '
'; }); el.innerHTML = html; }
